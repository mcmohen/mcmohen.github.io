 代码如下:
 ```cpp
#include <cstdio>
#include <iostream>
#include <cstring>
using namespace std;
int n,t,tem=0;  //题目中出现的变量
int a[50001];  //两个数组一个存数，一个存下标
int b[50001];
int main()
{
    cin>>t;
    for(int i=1;i<=t;i++)
    {
        cin>>n;
        for(int j=1;j<=n;j++)
        {
            cin>>tem;
            if(a[tem]==0)  //如果数组没存过值
            {
                a[tem]=tem;  //这里是最关键的
                b[j]=tem;  //a存值，b存下标
            }
        }
        for(int x=1;x<=n;x++)
        {
            if(b[x]!=0)  //输出到0就stop
            {
                cout<<b[x]<<' ';
            }
        }
        printf("\n");
        memset(b,0,sizeof(b));  //一定要初始化！
        memset(a,0,sizeof(a));
    }
} 
 ```
